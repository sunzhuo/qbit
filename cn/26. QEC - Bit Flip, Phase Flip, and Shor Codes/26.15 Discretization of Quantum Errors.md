# 量子误差的离散化

实现量子误差校正的关键思想是量子误差的离散化：能够使用明确允许我们仅校正一组离散泡利误差的代码来校正任何类型的错误。

> 离散量子误差能力背后的高级数学推理如下。
>
> 让我们考虑作用于编码状态的量子位之一的单量子位错误，将整个状态从 $\ket{\psi_L}$ 转换为 $E\ket{\psi_L}$。
> 您可以将具有此错误的状态表示为四个正交状态的某种叠加：$\ket{\psi_L}$、$X\ket{\psi_L}$、$Z\ket{\psi_L}$ 和 $XZ\ket{\psi_L}$（这是可能的，因为泡利矩阵 $I, X, Y, Z$ 形成 $2 \times 2$ Hermitian 矩阵的基础，因此表示的运算符误差可以分解为它们的线性组合）。
>
> 执行一组奇偶校验测量来检测 $X$ 和 $Z$ 错误会将状态 $E\ket{\psi_L}$ 折叠为这四种状态中的其他状态，从而产生与该状态匹配的测量结果。例如，如果奇偶校验测量产生一组结果，表明发生了 $X$ 错误，则状态 $E\ket{\psi_L}$ 将崩溃为 $X\ket{\psi_L}$。
>
> 然后，您可以使用奇偶校验测量的结果来纠正它们指示的错误，使状态返回到无错误状态 $\ket{\psi_L}$。

短码是否确实纠正了所有错误，而不仅仅是泡利错误集 $X$、$Y$ 和 $Z$？
我们来尝试一下吧！

以下演示汇总了使用 Shor 码进行纠错的步骤：它将给定的逻辑状态编码为多个量子位，引入任意错误，运行错误检测代码并在必要时应用纠错，并检查结果是否是起始逻辑状态的准确编码。尝试将不同的错误应用于代码的不同量子位 - 不仅是泡利错误，还包括任何单量子位旋转。您甚至可以使用测量！

来源：ShorCodeDemo.qs

```qsharp
namespace Kata {
    import Std.Arrays.*;
    import Std.Diagnostics.*;
    import Std.Math.*;

    @EntryPoint()
    operation RunExample() : Unit {
        // Allocate qubits and prepare initial logical state.
        use qs = Qubit[9];
        Ry(ArcCos(0.6), qs[0]);

        // Encode the logical state using Shor code.
        ShorEncode(qs);

        // Introduce an arbitrary error on one qubit.
        Rz(0.1, qs[5]);

        // Detect and correct error.
        let (ind, err) = ShorDetectError(qs);

        if ind > -1 {
            if err == PauliX {
                X(qs[ind]);
            } elif err == PauliY {
                Y(qs[ind]);
            } else {
                ForEach(q => Z(q), qs[ind * 3..ind * 3 + 2]);
            }
        }

        // Check that the state was error-corrected accurately by uncomputing
        // the encoding and state preparation to see that the result is the |0⟩ state.
        Adjoint ShorEncode(qs);
        Ry(-ArcCos(0.6), qs[0]);
        if not CheckAllZero(qs) {
            Message("State recovered incorrectly!");
            ResetAll(qs);
        } else {
            Message("State recovered correctly!");
        }
    }

    operation ShorEncode(qs : Qubit[]) : Unit is Adj + Ctl {
        BitflipEncode(qs[0..3..8]);
        ApplyToEachCA(H, qs[0..3..8]);
        for i in 0..2 {
            BitflipEncode(qs[3 * i..3 * i + 2]);
        }
    }

    operation BitflipEncode(qs : Qubit[]) : Unit is Adj + Ctl {
        CNOT(qs[0], qs[1]);
        CNOT(qs[0], qs[2]);
    }

    operation ShorDetectError(qs : Qubit[]) : (Int, Pauli) {
        // Detect X error first
        mutable x_ind = -1;
        for i in 0..2 {
            let err_ind = BitflipDetectError(qs[3 * i..3 * i + 2]);
            if err_ind > -1 {
                set x_ind = 3 * i + err_ind;
            }
        }

        // Detect Z error
        mutable z_ind = -1;
        let m1 = Measure([PauliX, size = 6], qs[0..5]);
        let m2 = Measure([PauliX, size = 6], qs[3..8]);

        if m1 == Zero and m2 == Zero {
            set z_ind = -1;
        } elif m1 == One and m2 == Zero {
            set z_ind = 0;
        } elif m1 == One and m2 == One {
            set z_ind = 1;
        } elif m1 == Zero and m2 == One {
            set z_ind = 2;
        }

        // Combine both errors into return value
        if x_ind == -1 and z_ind == -1 {
            return (-1, PauliI);
        }
        if x_ind > -1 and z_ind > -1 {
            return (x_ind, PauliY);
        }
        if x_ind > -1 {
            return (x_ind, PauliX);
        }
        return (z_ind, PauliZ);
    }

    operation BitflipDetectError(qs : Qubit[]) : Int {
        let m1 = Measure([PauliZ, PauliZ], qs[0..1]);
        let m2 = Measure([PauliZ, PauliZ], qs[1..2]);

        if m1 == One and m2 == Zero {
            return 0;
        } elif m1 == One and m2 == One {
            return 1;
        } elif m1 == Zero and m2 == One {
            return 2;
        } else {
            return -1;
        }
    }
}
```
